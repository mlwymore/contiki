00072 {
00073   StStatus status = ST_SUCCESS;
00074   int32u startCount;
00075   
00076   //There is really no reason to bother with a duration of 0ms
00077   if(*duration==0) {
00078     INTERRUPTS_ON();
00079     return status;
00080   }
00081   
00082   ATOMIC(
00083     //disable top-level interrupt while configuring
00084     INT_CFGCLR = INT_SLEEPTMR;
00085     
00086     //Our tick is calibrated to 1024Hz, giving a tick of 976.6us and an
00087     //overflow of 4194304.0 seconds.  Since our duration parameter is 32bits
00088     //and the sleep timer is also 32bits, we do not need to calculate or
00089     //handle any overflows, just handle the final compare value which is
00090     //current time plus the duration.
00091     startCount = halCommonGetInt32uMillisecondTick();
00092     SLEEPTMR_CMPAL = (startCount + (*duration))&0xFFFF;
00093     SLEEPTMR_CMPAH = ((startCount + (*duration))>>16)&0xFFFF;
00094     
00095     sleepTimerInterruptOccurred = FALSE;
00096     
00097     //clear any stale interrupt flag and set the CMPA interrupt
00098     INT_SLEEPTMRFLAG = INT_SLEEPTMRCMPA;
00099     INT_SLEEPTMRCFG = INT_SLEEPTMRCMPA;
00100     
00101     //enable top-level interrupt
00102     INT_CFGSET = INT_SLEEPTMR;
00103   )
00104   
00105   halSleep(SLEEPMODE_IDLE);
00106   
00107   INT_SLEEPTMRCFG = INT_SLEEPTMRCFG_RESET; //disable all SleepTMR interrupts
00108   
00109   if(!sleepTimerInterruptOccurred) {
00110     status = ST_SLEEP_INTERRUPTED;
00111     //Update duration to account for how long the sleep was
00112     *duration -= halCommonGetInt32uMillisecondTick() - startCount;
00113   } else {
00114     //since the sleepTimer interrupt triggered, we have slept for the entire
00115     //duration.  Set the duration to zero and return.

00124 #define CONVERT_TICKS_TO_QS(x) (x >> 8)
00125 #define TIMER_MAX_QS           0x1000000 // = 4194304 seconds * 4 = 16777216
00126 static StStatus internalSleepForQs(boolean useGpioWakeMask,
00127                                       int32u *duration,
00128                                       int32u gpioWakeBitMask)
00129 {
00130   StStatus status = ST_SUCCESS;
00131   int32u sleepOverflowCount;
00132   int32u remainder;
00133   int32u startCount;
00134   
00135   //There is really no reason to bother with a duration of 0qs
00136   if(*duration==0) {
00137     INTERRUPTS_ON();
00138     return status;
00139   }
00140   
00141   ATOMIC(
00142     //disable top-level interrupt while configuring
00143     INT_CFGCLR = INT_SLEEPTMR;
00144     
00145     //Our tick is calibrated to 1024Hz, giving a tick of 976.6us and an
00146     //overflow of 4194304.0 seconds.  Calculate the number of sleep overflows
00147     //in the duration
00148     sleepOverflowCount = (*duration)/TIMER_MAX_QS;
00149     //calculate the remaining ticks
00150     remainder = CONVERT_QS_TO_TICKS((*duration)%TIMER_MAX_QS);
00151     //grab the starting sleep count
00152     startCount = halCommonGetInt32uMillisecondTick();
00153     
00154     sleepTimerInterruptOccurred = FALSE;
00155     
00156     if(remainder) {
00157       //set CMPA value
00158       SLEEPTMR_CMPAL = (startCount+remainder)&0xFFFF;
00159       SLEEPTMR_CMPAH = ((startCount+remainder)>>16)&0xFFFF;
00160       //clear any stale interrupt flag and set the CMPA interrupt
00161       INT_SLEEPTMRFLAG = INT_SLEEPTMRCMPA;
00162       INT_SLEEPTMRCFG = INT_SLEEPTMRCMPA;
00163     }
00164     if(sleepOverflowCount) {
00165       //set CMPB value
00166       SLEEPTMR_CMPBL = startCount&0xFFFF;
00167       SLEEPTMR_CMPBH = (startCount>>16)&0xFFFF;
00168       //clear any stale interrupt flag and set the CMPB interrupt
00169       //this will also disable the CMPA interrupt, since we only want to wake
00170       //on overflows, not the remainder yet
00171       INT_SLEEPTMRFLAG = INT_SLEEPTMRCMPB;
00172       INT_SLEEPTMRCFG = INT_SLEEPTMRCMPB;
00173     }
00174     
00175     //enable top-level interrupt
00176     INT_CFGSET = INT_SLEEPTMR;
00177   )
00178   
00179   while(*duration > 0) {
00180     //halSleep doesn't exist in the minimal HAL.
00181     #ifndef MINIMAL_HAL
00182     if(!useGpioWakeMask) {
00183       halSleep(SLEEPMODE_WAKETIMER);
00184     } else
00185     #endif //MINIMAL_HAL
00186     {
00187       halSleepWithOptions(SLEEPMODE_WAKETIMER, gpioWakeBitMask);
00188     }
00189     
00190     INT_SLEEPTMRCFG = INT_SLEEPTMRCFG_RESET; //disable all SleepTMR interrupts
00191     
00192     //If we didn't come out of sleep via a compare or overflow interrupt,
00193     //it was an abnormal sleep interruption; report the event.
00194     if(!sleepTimerInterruptOccurred) {
00195       status = ST_SLEEP_INTERRUPTED;
00196       //Update duration to account for how long last sleep was.  Using the
00197       //startCount variable is always valid because full timer wraps always
00198       //return to this value and the remainder is an offset from this value.
00199       //And since the duration is always reduced after each full timer wrap,
00200       //we only need to calculate the final duration here.
00201       *duration -= CONVERT_TICKS_TO_QS(halCommonGetInt32uMillisecondTick() -
00202                                        startCount);
00203       break;
00204     } else {
00205       if(sleepOverflowCount) {
00206         sleepOverflowCount--;
00207         *duration -= TIMER_MAX_QS;
00208       } else {
00209         *duration -= CONVERT_TICKS_TO_QS(remainder);
00210       }
00211       sleepTimerInterruptOccurred = FALSE;
00212       if(sleepOverflowCount) {
00213         //enable sleeping for a full timer wrap
00214         INT_SLEEPTMRFLAG = INT_SLEEPTMRCMPB;
00215         INT_SLEEPTMRCFG = INT_SLEEPTMRCMPB;
00216       } else if(!sleepOverflowCount && (*duration>0)){
00217         //enable sleeping for the remainder
00218         INT_SLEEPTMRFLAG = INT_SLEEPTMRCMPA;
00219         INT_SLEEPTMRCFG = INT_SLEEPTMRCMPA;
00220       }
00221     }
00222   }
00223   
00224   return status;
00225 }
