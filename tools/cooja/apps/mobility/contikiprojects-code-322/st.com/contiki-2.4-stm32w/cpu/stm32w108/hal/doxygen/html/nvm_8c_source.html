00032 static int8u determineState(void)
00033 {
00034   int32u leftMgmt = *(int32u *)NVM_LEFT_PAGE;
00035   int32u rightMgmt = *(int32u *)NVM_RIGHT_PAGE;
00036   int8u state=0;
00037   
00038   if((leftMgmt==0xFFFF0000) && (rightMgmt==0xFFFFFFFF)) {
00039     //State 1 and state 4 use identical mgmt words.  The function
00040     //determineState() is only called at the start of a NVM read
00041     //or write.  During a read, state 1 and 4 both read from the
00042     //LEFT so there is no reason to make a distinction.  During
00043     //a write, the system will see the current page as LEFT and
00044     //therefore be transitioning from LEFT to RIGHT so state 4 is
00045     //correct.  State 1 is only required to transition from 0 to 2.
00046     state = 4;
00047   } else if((leftMgmt==0xFFFF0000) && (rightMgmt==0xFF00FFFF)) {
00048     state = 2;
00049   } else if((leftMgmt==0xFFFF0000) && (rightMgmt==0xFF000000)) {
00050     state = 3;
00051   } else if((leftMgmt==0xFFFF0000) && (rightMgmt==0xFFFFFFFF)) {
00052     state = 4;
00053   } else if((leftMgmt==0xFFFF0000) && (rightMgmt==0xFFFFFF00)) {
00054     state = 5;
00055   } else if((leftMgmt==0xFF000000) && (rightMgmt==0xFFFFFF00)) {
00056     state = 6;
00057   } else if((leftMgmt==0xFF000000) && (rightMgmt==0xFFFF0000)) {
00058     state = 7;
00059   } else if((leftMgmt==0xFFFFFFFF) && (rightMgmt==0xFFFF0000)) {
00060     state = 8;
00061   } else if((leftMgmt==0xFFFFFF00) && (rightMgmt==0xFFFF0000)) {
00062     state = 9;
00063   } else if((leftMgmt==0xFFFFFF00) && (rightMgmt==0xFF000000)) {
00064     state = 10;
00065   } else {
00066     //State 0 is used to indicate erased or invalid.
00067     state = 0;
00068   }
00069   
00070   return state;
00071 }

00090   assert(offset+length<NVM_DATA_SIZE_B);
00091   
00092   //Obtain the data from NVM storage.
00093   switch(determineState()) {
00094     case 1:
00095     case 2:
00096     case 3:
00097     case 4:
00098     case 9:
00099     case 10:
00100       flash = (int16u *)(NVM_LEFT_PAGE+offset);
00101       for(i=0;i<(length/2);i++) {
00102         ram[i] = flash[i];
00103       }
00104     break;
00105     case 5:
00106     case 6:
00107     case 7:
00108     case 8:
00109       flash = (int16u *)(NVM_RIGHT_PAGE+offset);
00110       for(i=0;i<(length/2);i++) {
00111         ram[i] = flash[i];
00112       }
00113     break;
00114     case 0:
00115     default:
00116       //Reading from NVM while the mgmt bytes are in an invalid state
00117       //should not return any bytes actually found in flash.  Instead,
00118       //return nothing but 0xFF.  This is legitimate because the next
00119       //call to the write function will also find invalid mgmt bytes
00120       //and trigger an erasure of NVM, after which the NVM really will
00121       //contain just 0xFF for data (plus the new data supplied during
00122       //the write call).
00123       for(i=0;i<(length/2);i++) {
00124         ram[i] = 0xFFFF;
00125       }
00126     //Inform the calling code. using ST_ERR_FATAL, that there were
00127     //invalid mgmt bytes and 0xFF was forcefully returned.
00128     return ST_ERR_FATAL;
00129   }

00131   return ST_SUCCESS;

00135 static int8u erasePage(int32u page)
00136 {
00137   StStatus status;
00138   int32u i, k;
00139   int32u address;
00140   int8u *flash;
00141   
00142   //Erasing a LEFT or RIGHT page requires erasing all of the flash pages.
00143   //Since the mgmt bytes are stored at the bottom of a page, the flash pages
00144   //are erased from the top down ensuring that that mgmt words are the last
00145   //data to be erased.  This way, if a reset occurs while erasing, the mgmt
00146   //words are still valid the next time determineState() is called.
00147   for(i=NVM_FLASH_PAGE_COUNT;i>0;i--) {
00148     address = (page+((i-1)*MFB_PAGE_SIZE_B));
00149     flash = (int8u *)address;
00150     //Scan the page to determine if it is fully erased already.
00151     //If the flash is not erased, erase it.  The purpose of scanning
00152     //first is to save a little time if erasing is not required.
00153     for(k=0;k<MFB_PAGE_SIZE_B;k++,flash++) {
00154       if(*flash != 0xFF) {
00155         status = halInternalFlashErase(MFB_PAGE_ERASE, address);
00156         if(status != ST_SUCCESS) {
00157           return status;
00158         }
00159         //Don't bother looking at the rest of this flash page and just
00160         //move to the next.
00161         k=MFB_PAGE_SIZE_B;
00162       }
00163     }
00164   }
00165   return ST_SUCCESS;
00166 }
